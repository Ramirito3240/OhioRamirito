    local function getClosestPlayerToCursor(trg_part)
        local nearest = nil
        local last = math.huge
        for _, v in pairs(Players:GetPlayers()) do
            if v ~= Players.LocalPlayer and v.Character and v.Character:FindFirstChild(trg_part) then
                if Players.LocalPlayer.Character:FindFirstChild(trg_part) then
                    local ePos, vissss = Camera:WorldToViewportPoint(v.Character[trg_part].Position)
